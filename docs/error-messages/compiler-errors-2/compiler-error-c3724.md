# <a name="compiler-error-c3724"></a>C3724 de erro do compilador
deve #include \<Windows. h > para usar multi-Threading com eventos  
  
 O arquivo do Windows. h é necessário se você usar multi-Threading com eventos. Para corrigir esse erro, adicione `#include <windows.h>` na parte superior do arquivo no qual fontes de evento e eventos receptores são definidos.  
  
```  
// C3724.cpp  
// uncomment the following line to resolve  
// #include <windows.h>  
  
[event_source(native), threading(apartment)]  
class CEventSrc {  
public:  
   __event void event1();   // C3724  
};  
  
[event_receiver(native)]  
class CEventRec {  
public:  
   void handler1() {  
   }  
  
   void HookEvents(CEventSrc* pSrc) {  
      __hook(CEventSrc::event1, pSrc, CEventRec::handler1);  
   }  
  
   void UnhookEvents(CEventSrc* pSrc) {  
      __unhook(CEventSrc::event1, pSrc, CEventRec::handler1);  
   }  
};  
  
int main() {  
}  
```
